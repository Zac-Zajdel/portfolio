# Filter using `match()` function

## Code

```jsx
const frameworks = [
  { name: "React" },
  { name: "Vue" },
]

const [filteredFrameworks, setFilteredFrameworks] = useState(frameworks)

const handleChange = (event: HTMLInputElement) => {
  const input = event.target.value.toLowerCase()
  setFilteredFrameworks(
    frameworks.filter((framwork) => framework.name.toLowerCase().match(input))
  )
}
```

## Usage
```jsx
<div className="relative max-w-lg">
  <input
    aria-label="Search frameworks"
    type="text"
    placeholder="Search frameworks"
    className="block w-full rounded-md border border-gray-300 dark:bg-white dark:bg-gray-800 px-4 py-2 text-gray-900 dark:border-gray-900 dark:text-gray-100"
    onChange={handleChange}
  />
```
